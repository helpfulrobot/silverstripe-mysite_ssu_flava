\HTMLEditorFieldSmall

Sometimes you want two or more TinyMCE instances in the same page, and for each one to have it's own tinymce configuration. This class lets you do just that.

To use, pop something like this into your _config.php (this is a copy of what's in cms/_config, modified for
our purposes)

mlEditorConfig::get('linksonly')->setOptions(array(
riendly_name' => 'Links Only',
ode' => 'none',
anguage' => i18n::get_tinymce_lang(),

ody_class' => 'typography',
ocument_base_url' => Director::absoluteBaseURL(),

rlconverter_callback' => "nullConverter",
etupcontent_callback' => "sapphiremce_setupcontent",
leanup_callback' => "sapphiremce_cleanup",

se_native_selects' => true, // fancy selects are bug as of SS 2.3.0
alid_elements' => "@[id|class|style|title],#a[id|rel|rev|dir|tabindex|accesskey|type|name|href|target|title|class]",
xtended_valid_elements' => "",

utton_tile_map' => true
;

mlEditorConfig::get('linksonly')->enablePlugins(array('ssbuttons' => '../../../cms/javascript/tinymce_ssbuttons/editor_plugin_src.js'));

mlEditorConfig::get('linksonly')->setButtonsForLine(1, array('sslink'));
mlEditorConfig::get('linksonly')->setButtonsForLine(2, array());
mlEditorConfig::get('linksonly')->setButtonsForLine(3, array());
